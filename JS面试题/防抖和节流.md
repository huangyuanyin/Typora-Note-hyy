###### 防抖或是节流：限制函数的执行次数。

1. 防抖：通过setTimeout的方式，在一定的时间间隔内，将多次触发变成一次触发；

2. 节流：减少一段时间的触发频率；

   

> 参考博客：[一篇文章掌握性能优化大法之防抖和节流](https://juejin.cn/post/7018341638451757087)



### 防抖代码：

```javascript
// eg1：点击事件
<div id="app">
      <input type="text">
      <input type="submit" id="input">
</div>
<script type="text/javascript">
      var btn = document.getElementById('input')
	  btn.addEventListener('click',debounce(submit,1000),false)      
	  // btn.addEventListener('click',submit,false)
 	  
      function submit(){
        console.log(1);
        // console.log(this); // this指向问题
        // console.log(e)       // 此时的事件参数e
      }

      function debounce(fn,timer){
        var time = null
        return function(){
          var firstClick = !time  // true

          if(time){ clearTimeout(time) }

          if(firstClick){  // 判断是否是第一次点击
            fn.apply(this,arguments)
          }
            
          time = setTimeout(() => {
            time = null
          },timer)
        }
      }
</script>
```

```javascript
// eg2:输入事件
<div id="app">
    <input type="text">
</div>
<script type="text/javascript">
    let inp = document.querySelector("input")
    inp.oninput = debounce(submit,1000)
    function submit(){
      console.log(this.value);
    }
    function debounce(fn,timer){
      let time = null
      return function(){
      	if (time !== null) {
            clearTimeout(time)
        }
        time = setTimeout(()=>{
            // fn.apply(this,arguments)
            fn.call(this)
        },timer)
      }
 	}
</script>
```



### 节流代码:

```javascript
<div id="app">
      <input type="text">
      <input type="submit" id="input">
</div>
<script type="text/javascript">
      var btn = document.getElementById('input')
	  btn.addEventListener('click',trottle(submit,1000),false)      
 	  
      function submit(){
        console.log(1);
      }

	  function trottle(fn,timer){
        var begin = 0;
          
        return function(){
            
          var cur = new Date().getTime() // 点击按钮时的时间戳
          console.log(cur - begin);
            
          if (cur - begin > timer) {
            fn.apply(this,arguments)
            begin = cur
          }
        }
      }
</script>
```
